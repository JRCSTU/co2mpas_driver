@sh.add_function(dsp, outputs=['discrete_poly_spline'])
def define_discrete_poly(poly_spline, sp_bins):
    """
    Define discrete poly.

    :param poly_spline:
        Poly spline.
    :type poly_spline:

    :param sp_bins:
        Speed bins.
    :type sp_bins:

    :return:
    """
    return [acc(sp_bins) for acc in poly_spline]
